使用闭区间的二分搜索

```rust
impl Solution {
    pub fn my_sqrt(x: i32) -> i32 {
        return Solution::my_sqrt_i64(x as i64);
    }
    pub fn my_sqrt_i64(x: i64) -> i32 {
        let mut i = 0;
        let mut j = x;
        while i < j {
            let m = i + (j - i + 1) / 2;
            if m * m == x {
                return m as i32;
            }
            if m * m < x {
                i = m;
            } else {
                j = m - 1;
            }
        }
        return i as i32;
    }
}
```
