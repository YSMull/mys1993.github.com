class Solution {
    public int[] searchRange(int[] nums, int target) {
        int l = 0, r = nums.length;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (nums[mid] >= target) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        // 二分结束
        if (l == nums.length // 数组中没有满足性质的点
            || nums[l] != target // 满足性质的点并不是 target
        ) {
            return new int[]{-1, -1};
        }
        // 此时 l 是第一个等于 target 的点
        int l2 = l;
        // 处理 l 已经是最后一个点的情况
        for ( ; l2 < nums.length - 1 && nums[l] == nums[l2 + 1] ; l2++);
        return new int[]{l, l2};
    }
}
